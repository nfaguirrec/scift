402  function var( this, weights ) result( output )
403  class(realhistogram), intent(in) :: this
404  real(8), optional, intent(in) :: weights(:)
405  real(8) :: output
406 
407  class(reallistiterator), pointer :: iter
408  real(8) :: mean
409  integer :: i
410 
411  if( this.size() == 1 ) then
412  output = 0.0_8
413  return
414  end if
415 
416  if( present(weights) ) then
417  if( size(weights) < this.size() ) then
418  write(*,"(A,I5,A,I5,A)") "### ERROR ### RealHistogram.var(). Inconsistent size for weights. size(weights) /= this.size() (", size(weights), ", ", this.size(), ")"
419  stop
420  end if
421 
422  mean = this.mean( weights )
423 
424  output = 0.0_8
425  iter => this.begin
426  i = 1
427  do while( associated(iter) )
428  output = output + weights(i)*( iter.data - mean )**2
429 
430  iter => iter.next
431  i = i+1
432  end do
433 
434  ! Approximation of the unbiased weighted covariance matrix, but without Bessel correction
435  output = output/sum(weights(1:i-1))
436  else
437  mean = this.mean()
438 
439  output = 0.0_8
440  iter => this.begin
441  do while( associated(iter) )
442  output = output + ( iter.data - mean )**2
443 
444  iter => iter.next
445  end do
446 
447  ! Bessel's correction
448  ! n->n-1
449  ! standard deviation of the sample (considered as the entire population) -> Corrected sample standard deviation
450  ! Standard deviation of the population -> sample standard deviation
451  ! sigma -> s
452  output = output/real(this.size()-1,8)
453  end if
454  end function var
