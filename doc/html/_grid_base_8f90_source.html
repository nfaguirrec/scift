201 
202  if( columneff <= size(tokens) ) then
203  if( len(trim(tokens(columneff))) /= 0 ) then
204  read( tokens(columneff),* ) data(ndata)
205  ndata = ndata + 1
206  end if
207  end if
208  end do
209

218  this.max = maxval(this.data)
219  this.size = size(this.data)
220 
221  effstepsize = this.data(2)-this.data(1)
222  do i=3,size(this.data)
