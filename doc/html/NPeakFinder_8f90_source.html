334  function execute( this, posVec ) result( output )
335  class(npeakfinder) :: this
336  type(integerlist), optional :: posvec ! Peak positions in this.func
337  type(rnfunction) :: output
338 
339  type(integerlist) :: effposvec
340 
341  integer :: k, h, w
342  real(8) :: m, s
343  integer :: npoints
344 
345  integer :: i
346  real(8), allocatable :: a(:)
347  real(8), allocatable :: x(:), y(:)
348 
349  k = this.windowsize
350  h = this.tolerance
351  w = this.bandwidth
352 
353  npoints = this.func.xgrid.npoints
354 
355  allocate( a(npoints) )
356  call effposvec.init()
357 
358  !------------------------------------------------------------
359  ! Compute peak function value for each of the N points in T
360  !------------------------------------------------------------
361  do i=1,npoints
362  select case( this.method )
363  case( npeakfinder_max_dist )
364  a(i) = kernels1(k,i,this.func.farray)
365  case( npeakfinder_max_aver_neig )
366  a(i) = kernels2(k,i,this.func.farray)
367  case( npeakfinder_max_aver_neig_aver )
368  a(i) = kernels3(k,i,this.func.farray)
369  case( npeakfinder_entropy )
370  a(i) = kernels4(k,w,i,this.func.farray)
371  case default
372  write(*,*) "### ERROR ### PeakFinder.execute. Method XXX is not implemented yet"
373  stop
374  end select
375  end do
376 
377  !-------------------------------------------------
378  ! Compute the mean m and standard deviation s of
379  ! all positive values in array a
380  !-------------------------------------------------
381  m = sum( a(k+1:npoints-k), mask=a(k+1:npoints-k)>0.0_8 )/count( a(k+1:npoints-k)>0.0_8 )
382  s = sqrt( sum( (a(k+1:npoints-k)-m)**2, mask=a(k+1:npoints-k)>0.0_8 )/count( a(k+1:npoints-k)>0.0_8 ) )
383 
384  !--------------------------------------
385  ! Remove local peaks which are small
386  ! in global context
387  !--------------------------------------
388  do i=1,npoints
389  if( a(i)>0.0_8 .and. (a(i)-m) > h*s ) then
390  call effposvec.append( i )
391  end if
392  end do
393 
394  !----------------------------------------------
395  ! Retain only one peak out of any set of peaks
396  ! within distance k of each other
397  !----------------------------------------------
398  i=2
399  do while( i<=effposvec.size() )
400  if( abs( effposvec.at(i)-effposvec.at(i-1) ) <= k ) then
401  if( this.func.at( effposvec.at(i) ) < this.func.at( effposvec.at(i-1) ) ) then
402  call effposvec.erase( i )
403  else
404  call effposvec.erase( i-1 )
405  end if
406  i=2
407  else
408  i=i+1
409  end if
410  end do
411 
412  !----------------------------
413  ! Build the output function
414  !----------------------------
415  if( present(posvec) ) then
416  posvec = effposvec
417  end if
418 
419  allocate( x(effposvec.size()) )
420  allocate( y(effposvec.size()) )
421 
422  do i=1,effposvec.size()
423  x(i) = this.func.xgrid.at( effposvec.at(i) )
424  y(i) = this.func.at( effposvec.at(i) )
425  end do
426 
427  call output.init( x, y )
428 
429  deallocate( x )
430  deallocate( y )
431  end function execute

446  function npeakfinder_generatesignal( tMin, tMax, dt, nGauss, SNFactor, maxI, minDesv, maxDesv ) result( output )
447  real(8), intent(in) :: tmin
448  real(8), intent(in) :: tmax
449  real(8), intent(in) :: dt
450  integer, intent(in) :: ngauss
451  real(8), intent(in) :: snfactor
452  real(8), intent(in) :: maxi
453  real(8), intent(in) :: mindesv, maxdesv
454  type(rnfunction) :: output
455 
456  integer :: npoints
457  real(8) :: tn
458  integer :: n, m
459 
460  real(8), allocatable :: i(:)
461  real(8), allocatable :: t0(:)
462  real(8), allocatable :: s(:)
463  real(8) :: value
464  real(8) :: randnumber
465 
466  real(8), allocatable :: x(:), y(:)
467 
468  npoints = int( abs( tmax-tmin )/dt+1 )
469 
470  allocate( x(npoints), y(npoints) )
471  allocate( i(ngauss), t0(ngauss), s(ngauss) )
472 
473  call random_seed()
474  do n=1,ngauss
475  call random_number( randnumber )
476  i(n) = randnumber*maxi
477 
478  call random_number( randnumber )
479  t0(n) = tmin+randnumber*abs( tmax-tmin )
480 
481  call random_number( randnumber )
482  s(n) = mindesv+randnumber*abs( maxdesv-mindesv )
483  end do
484 
485  tn = tmin
486  do n=1,npoints
487  value = 0.0_8
488  do m=1,ngauss
489  value = value + gaussian( tn, i(m), t0(m), s(m) )
490  end do
491 
492  call random_number( randnumber )
493  value = value + ( snfactor*maxval(i)*( 2.0_8*randnumber-1.0_8 ) )
494  value = max( value, 0.0_8 )
495 
496  x(n) = tn
497  y(n) = cmplx( value, 0.0_8 )
498 
499  tn = tn + dt
500  end do
501 
502  call output.init( x, y )
503  end function npeakfinder_generatesignal
