143  !
144  !! weights c_{i,j}^k [c(i,j,k)] such that the approximations
145  !!
146  !! f^{(k)}(ksi) \approx \sum_{j=0}^i c_{i,j}^k f(x(j)),
147  !!
148  !! k=0,1,...,m, i = k, k+1,...,n
149  !!
150  !! are all optimal.
151  !!
152  !! C(i,j,k) is the weight to be applied at x(j) when the kth derivative is
153  !! approximated by a stencil extending over x(0), x(1), ..., x(i). (j <= i)
154  !!
155  !! The case m=0 provides the fastest way known for polynomial interpolation
156  !! at a single point.
157  !!
158  !! \param[in] x a zero-based array of reals; the grid points
159  !! \param[in] ksi a real; point at which the approximations are wanted
160  !! \param[in] m an integer; highest order of derivative of interest
161  !! \param[out] c tridimensional zero-based array of reals;
162  !! C(i,j,k) is the weight to be applied at x(j) when the kth derivative is
163  !! approximated by a stencil extending over x(0), x(1), ..., x(i). (j <= i)
164  !!
165  !! \return nothing
166  !!
167  subroutine fornbergweights( x, ksi, m, c )
168  integer, parameter :: dpk = kind(1.d0)
169 
170  real(dpk), intent(in) :: x(0:)
171  real(dpk), intent(in) :: ksi
172  integer, intent(in) :: m
173  real(dpk), intent(out) :: c(0:,0:,0:)
174 
175  real (dpk) :: a, b, temp1, temp2, d
176  integer :: i, j, k
177  integer :: minim
178 
179  c(0,0,0) = 1.0_dpk
180 
181  a = 1.0_dpk
182 
183  do i=1,ubound(x,1)
184  minim = min(i,m)
185  b = 1.0_dpk
186 
187  temp1 = x(i) - ksi
188 
189  do j=0,i-1
190  temp2 = x(i) - x(j)
191 
192  b = b * temp2
193 
194  if( i <= m ) c(i-1,j,i) = 0.0_dpk
195 
196  c(i,j,0) = temp1 * c(i-1,j,0) / temp2
197 
198  do k = 1, minim
199  c(i,j,k) = (temp1 * c(i-1,j,k) - k * c(i-1,j,k-1)) / temp2
200  enddo
201  enddo
202 
203  d = a / b
204  temp1 = d * (ksi - x(i-1))
205 
206  c(i,i,0) = temp1 * c(i-1,i-1,0)
207 
208  do k = 1, minim
209  c(i,i,k) = d * k * c(i-1,i-1,k-1) + temp1 * c(i-1,i-1,k)
210  enddo
211 
212  a = b
213  enddo
214 
215  end subroutine fornbergweights

223  function evaluateapoint( this, x, order ) result( output )
224  class(nderivator) :: this
225  real(8), intent(in) :: x
226  integer, intent(in) :: order
227  real(8) :: output
228 
229  integer :: i, ind
230  real(8), allocatable :: coeff(:,:,:)
231 
232  allocate( coeff(this.npoints,this.npoints,0:order) )
233 
234  if( this.func.xgrid.isequallyspaced ) then
235 
236  ind = (x-this.func.xgrid.min)/this.func.xgrid.stepsize+1
237 
238  else
239  do i=1,this.func.xgrid.npoints
240  if( x < this.func.xgrid.data(i) ) then
241  exit
242  end if
243  end do
244 
245  ind = i-1
246 
247 ! write(6,"(A)") "### ERROR ### Numerical derivatives for non equally spaced grids is not implemented"
248 ! stop
249  end if
250 
251  ! Diferencias hacia adelante
252  if( ind <= this.npoints/2 ) then
253  call fornbergweights( this.func.xgrid.data(1:this.npoints), x, order, coeff )
254  output = sum( coeff( this.npoints, :, order )*this.func.farray(1:this.npoints) )
255  ! Diferencias hacia atr√°s
256  else if( ind >= this.func.xgrid.npoints-this.npoints/2+1 ) then
257  call fornbergweights( this.func.xgrid.data(this.func.xgrid.npoints-this.npoints+1:this.func.xgrid.npoints), x, order, coeff )
258  output = sum( coeff( this.npoints, :, order )*this.func.farray(this.func.xgrid.npoints-this.npoints+1:this.func.xgrid.npoints) )
259  ! Diferencias centradas
260  else
261  call fornbergweights( this.func.xgrid.data(ind-this.npoints/2:ind+this.npoints/2), x, order, coeff )
262  output = sum( coeff( this.npoints, :, order )*this.func.farray(ind-this.npoints/2:ind+this.npoints/2) )
263  end if
264 
265  deallocate( coeff )
266  end function evaluateapoint
