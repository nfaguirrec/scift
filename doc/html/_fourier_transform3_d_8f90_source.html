143  subroutine init( this, iFunc, domain, oFunc )

146  integer(8), intent(in), optional :: domain

148 
149  integer(8) :: effdomain
150

156 
157  nx = this.npoints(1)
158  ny = this.npoints(2)
159  nz = this.npoints(3)
160 
161  effdomain = fouriertransform_spatial_domain
162  if( present(domain) ) effdomain = domain
163 
164  if( present(ofunc) ) call goptions_error( "oFunc parameter is not implamented yet", "FourierTransform3D.init()" )
165 
166  call dfftw_plan_dft_3d( this.planf, nx, ny, nz, this.ifunc.farray, this.ifunc.farray, fftw_forward, fftw_estimate )

168 
169  if( effdomain == fouriertransform_spatial_domain ) then
170 
171  this.x = this.ifunc.xyzgrid ! << Hay que cambiar para hacer la transformada de algunos puntos
172  this.omega = fouriertransform3d_omegagrid( this.x )
173 
174  else if( effdomain == fouriertransform_frequency_domain ) then
175 
176  call goptions_warning( "FourierTransform_FREQUENCY_DOMAIN have not been tested yet", "FourierTransform.init()" )

257  subroutine execute( this, sgn, sync, shift )
258  class(fouriertransform3d) :: this
259  integer, intent(in) :: sgn
260  logical, optional, intent(in) :: sync
261  logical, optional, intent(in) :: shift
262 
263  logical :: effsync
264  logical :: effshift
265 
266  effsync = .false.
267  if( present(sync) ) effsync = sync
268 
269  effshift = .false.
270  if( present(shift) ) effshift = shift
271 
272  if( sgn == fouriertransform_forward ) then
273 
274  call dfftw_execute( this.planf )
275 
276  if( effsync ) then
277  this.ifunc.xyzgrid = this.omega
278 
279  if( effshift ) then
280  call fouriertransform3d_phase( this.ifunc )
281  call fouriertransform3d_shift( this.ifunc )
282  end if
283  end if
284 
285  else if ( sgn == fouriertransform_backward ) then
286 
287  if( effsync ) then
288  if( effshift ) then
289  call fouriertransform3d_ishift( this.ifunc )
290  call fouriertransform3d_phase( this.ifunc )
291  end if
292  end if
293 
294  call dfftw_execute( this.planb )
295 
296  if( effsync ) then
297  this.ifunc.xyzgrid = this.x
298  end if
299 
300  this.ifunc = this.ifunc/real(this.nPoints(1)*this.nPoints(2)*this.nPoints(3),8)
301 
302  else
303  call goptions_error( "Bad value for sgn", "FourierTransform3D.execute()" )
304  end if
305  end subroutine execute
