483  function inverse2x2( this ) result( inv )
484  class(matrix) :: this
485  type(matrix) :: inv
486 
487  type(matrix) :: i
488  real(8) :: deta
489 
490  if( this.isdiagonal() ) then
491  call inv.init( 2, 2, val=0.0_8 )
492 
493  inv.data(1,1) = 1.0_8/this.data(1,1)
494  inv.data(2,2) = 1.0_8/this.data(2,2)
495 
496  return
497  end if
498 
499  deta = this.determinant2x2()
500 
501  if( abs(deta) < 1d-16 ) then
502  write(*,*) "### WARNING ### Matrix.inverse2x2: Singular matrix detected ( detA < 1e-16 )"
503  write(*,*) " It will try to fix by Tikhonov regularization ( lambda = 1d-3 )"
504  call i.identity( 2, 2 )
505  this = this + i*1d-3 ! Tikhonov regularization
506  end if
507 
508  if( abs(deta) < 1d-16 ) then
509  write(*,*) "### ERROR ### Matrix.inverse2x2: Singular matrix detected ( detA < 1e-16 )"
510  write(*,*) "A = "
511  call this.show( formatted=.true. )
512  write(*,*) ""
513  write(*,*) "det(A) = ", deta
514  stop
515  end if
516 
517  call inv.init( 2, 2 )
518 
519 #define a(i,j) this.data(i,j)
520 #define b(i,j) inv.data(i,j)
521  b(1,1) = a(2,2)
522  b(1,2) = -a(1,2)
523  b(2,1) = -a(2,1)
524  b(2,2) = a(1,1)
525 #undef a
526 #undef b
527 
528  inv = inv/deta
529  end function inverse2x2
